C# — HashSet O(n) algorithm
int[] arr = { 1,9,3,10,4,20,2 }; HashSet<int> set = new HashSet<int>(arr); // O(n) build int longest = 0; foreach (int num in arr) { // only start chain at the BEGINNING of a sequence if (!set.Contains(num - 1)) // num-1 not in set → start here { int current = num, count = 1; while (set.Contains(current + 1)) // extend as far as possible { current++; count++; } longest = Math.Max(longest, count); } } // Time: O(n) Space: O(n) Console.WriteLine("Longest = " + longest);
