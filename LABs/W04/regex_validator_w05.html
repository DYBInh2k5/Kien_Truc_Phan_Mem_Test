using System.Text.RegularExpressions; static bool ValidateEmail(string input) => Regex.IsMatch(input, @"^[\w.-]+@[\w.-]+\.\w+$"); static bool ValidateCCCD(string input) => Regex.IsMatch(input, @"^\d{12}$"); static bool ValidateFullname(string input) => Regex.IsMatch(input, @"^[A-ZÀ-Ỹ][a-zà-ỹ]+(\s[A-ZÀ-Ỹ][a-zà-ỹ]+)+$"); // Usage Console.WriteLine(ValidateEmail("nguyen@gmail.com")); // True Console.WriteLine(ValidateCCCD("001203012345")); // True Console.WriteLine(ValidateFullname("Nguyễn Văn An")); // True
