static int LIS(int[] arr) { int n = arr.Length; int[] dp = new int[n]; // dp[i] = LIS length ending at i int[] prev = new int[n]; // prev[i] = predecessor index (backtrack) for (int i = 0; i < n; i++) { dp[i] = 1; prev[i] = -1; } for (int i = 1; i < n; i++) for (int j = 0; j < i; j++) if (arr[i] > arr[j] && dp[i] < dp[j] + 1) { dp[i] = dp[j] + 1; // extend LIS from j prev[i] = j; // remember where we came from } int maxIdx = 0; for (int i = 0; i < n; i++) if (dp[i] > dp[maxIdx]) maxIdx = i; // backtrack to reconstruct sequence var seq = new List<int>(); for (int k = maxIdx; k >= 0; k = prev[k]) { seq.Insert(0, arr[k]); if (prev[k] == -1) break; } return dp[maxIdx]; // also return seq }
